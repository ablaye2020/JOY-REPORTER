async function lancerSignalement() { const numero = document.getElementById('numero').value; if(!numero || numero.length < 10) { alert("Numéro invalide !"); return; } document.getElementById('status').innerHTML = "⏳ Envoi des 50 signalements en parallèle..."; document.querySelector('button').disabled = true; // Créer 50 promesses (envois simultanés) const promesses = []; for(let i = 1; i <= 50; i++) { promesses.push( fetch('/api/signal', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ numero: numero, index: i }) }) ); } // Attendre que les 50 soient terminés const resultats = await Promise.all(promesses); const reussis = resultats.filter(r => r.ok).length; document.getElementById('status').innerHTML = `✅ ${reussis}/50 signalements envoyés en parallèle !`; document.querySelector('button').disabled = false; }
